```c
#include "stdio.h"
#include "unistd.h"
#include "pthread.h"

#define N 5

int sigId=32;
int espera=10;

pthread_t hilos[N];
pthread_mutex_t m;

void* siguienteId (void* p);
void* esperar (void* p);

int main (int argc, char *argv[]){

	/*
	siguienteId(NULL);
	siguienteId(NULL);
	siguienteId(NULL);
	siguienteId(NULL);
	*/

	pthread_mutex_init(&m,NULL);

	for(int i=0;i<N;i++){
		//pthread_create(&hilos[i],NULL,&siguienteId,NULL);
		pthread_create(&hilos[i],NULL,&esperar,NULL);
	}
	//sleep(1);
	void **retval;
	for(int i=0;i<N;i++){
		pthread_join(hilos[i],retval);
	}

	pthread_mutex_destroy(&m);

	return 0;
}

void* siguienteId (void* p){

	pthread_mutex_lock(&m);

	printf("%d\n",sigId);
	sigId++;

	pthread_mutex_unlock(&m);

	return 0;
}

void* esperar (void* p){

	if(pthread_mutex_trylock(&m)==0){
		while(espera>0){
			printf("%d\n",espera);
			sleep(1);
			espera--;
		}

		pthread_mutex_unlock(&m);
	}
	else{
		printf("El mutex ya estaba bloqueado :(\n");
	}

	return 0;
}
```
